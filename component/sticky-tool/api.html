export interface IExStickyItemProps {
  icon?: string | IScopedSlot;
  label?: string | IScopedSlot;
  trigger?: 'hover' | 'click';
  width?: string | number;
  height?: string | number;
  type?: 'normal' | 'compact';
  /** 弹出层内容，可自定义渲染 */
  popoverContent?: string | IScopedSlot
}
