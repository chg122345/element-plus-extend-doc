export interface IExFormSchema extends IExFormItemProps {
  prop?: string;
  label?: string;
  show?: boolean | ((params: IExFormItemParams) => boolean);
  disabled?: boolean | ((params: IExFormItemParams) => boolean);
  renderFormItem?: boolean | ((params: IExFormItemParams) => boolean);
  component?: IExFormSchemaComponent;
  componentProps?: ObjectConstructor | ((params: IExFormItemParams) => ObjectConstructor);
  defaultValue?: any;
  renderContent?: (params: IExFormItemParams) => IComponentNode;

}
